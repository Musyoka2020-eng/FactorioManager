    Module exports: DepType, DepState, DepNode, build_dep_graph

    DepType(str, Enum): REQUIRED="required", OPTIONAL="optional", INCOMPATIBLE="incompatible", EXPANSION="expansion"
    DepState(str, Enum): INSTALLED="installed", MISSING="missing", PORTAL_ONLY="portal_only", EXPANSION="expansion", CIRCULAR="circular"
    DepNode dataclass: name: str, dep_type: DepType, state: DepState, version_constraint: str="", installed_version: Optional[str]=None, children: list[DepNode]=field(default_factory=list)

    _parse_raw_dep(raw: str) -> tuple[str, DepType, str]:
    - Strip whitespace; skip if empty or name is "base"
    - "!" prefix → INCOMPATIBLE; "(?)"|"?" prefix → OPTIONAL; else → REQUIRED
    - Extract name = first token before any of: space / > / = / < / !
    - version_constraint = remainder after name, stripped
    - If name in FACTORIO_EXPANSIONS and dep_type == REQUIRED: override dep_type = EXPANSION

    _get_dep_strings(root_name, installed_mods, portal) -> list[str]:
    - If root_name in installed_mods and installed_mods[root_name].raw_data.get("releases"): use releases[-1]["info_json"].get("dependencies", [])
    - Elif root_name in installed_mods: reconstruct list from mod.dependencies + ["? "+n for n in mod.optional_dependencies] + ["! "+n for n in mod.incompatible_dependencies] + list(mod.expansion_dependencies)
    - Else: call portal.get_mod(root_name) and use releases[-1]["info_json"]["dependencies"]; on PortalAPIError return []

    build_dep_graph(root_name, installed_mods, portal, *, full=False, _visited=None, _depth=0) -> list[DepNode]:
    - _visited = set() if _visited is None
    - Add root_name to _visited before processing
    - Get dep strings via _get_dep_strings
    - For each dep string: parse with _parse_raw_dep; skip empty names and "base"
    - Determine state:
      - dep_type == EXPANSION → state = DepState.EXPANSION, no recursion
      - dep_type == INCOMPATIBLE → state = DepState.INSTALLED if name in installed_mods else DepState.PORTAL_ONLY (incompatible = already-installed conflict; not-installed = no conflict)
      - name in installed_mods → state = DepState.INSTALLED
      - NOT in installed_mods AND full=False → state = DepState.PORTAL_ONLY (no recursion)
      - NOT in installed_mods AND full=True AND name in _visited → state = DepState.CIRCULAR, no recursion
      - NOT in installed_mods AND full=True AND _depth >= 2 → state = DepState.PORTAL_ONLY (depth cap)
      - NOT in installed_mods AND full=True: try portal.get_mod(name) to confirm existence; PortalAPIError(not_found) → MISSING; other PortalAPIError → PORTAL_ONLY
    - Recursion: only when full=True AND state == INSTALLED AND name not in _visited AND _depth < 2; recurse with _visited=_visited, _depth=_depth+1
    - installed_version = installed_mods[name].version if name in installed_mods else None
    - Return flat list of DepNode (all dep types mixed — UI groups them by dep_type)
  </behavior>
